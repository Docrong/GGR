var dhxWins,
w1,
menu;
function doOnLoad() {
    dhxWins = new dhtmlXWindows();
    dhxWins.enableAutoViewport(false);
    dhxWins.attachViewportTo("winVP");
    dhxWins.setImagePath("../../codebase/imgs/");
    w1 = dhxWins.createWindow("w1", 20, 30, 400, 350);
    w1.setText("Attach dhtmlxMenu to Modal Window");
    w1.setModal(true);
    w1.attachURL("../common/attach_url_ajax_inner.html", true);
    menu = w1.attachMenu();
    menu.setImagePath("../../../dhtmlxMenu/codebase/imgs/");
    menu.setIconsPath("../../../dhtmlxMenu/samples/common/imgs/");
    menu.loadXML("../../../dhtmlxMenu/samples/common/dhxmenu.xml?" + new Date().getTime());
}
</script>
